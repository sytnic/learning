## 009-Dialect nuances in MySQL

```sql
-- nuances MySQL

-- reviewing CRUD (create, read, update, delete)

-- create
insert into public.customers
values (
79,'Amata', 'Lee', 'amata@test.com', '111-111-1111', 'home','New York', 'NY', '00000'
)

-- read
select * from customers

-- update
UPDATE Customers
SET first_name = 'Lee', last_name= 'Amata'
WHERE id = 79;

/* delete statement */
DELETE FROM customers
WHERE id = 79;

-- case sensitive: MySQL is not case sensitive
select category
from products
where category = 'Desserts'
-- or
select category
from products
where category = 'dessertS'

-- case/when versus if/null
select
ifnull (first_name,'ERROR')
from customers
-- В строках first_name, где есть NULL,
-- будет вставлено 'ERROR' 

select
case when first_name is null
   then 'Error!'
   else first_name
end as first_name
from customers
-- Тот же самый результат.
-- Case чаще, чем ifnull используется в других БД.

```
