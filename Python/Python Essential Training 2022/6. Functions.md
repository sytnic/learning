Named Parameters

```py
def performOperation(num1, num2, operation='sum'):
    if operation == 'sum':
        return num1 + num2
    if operation == 'multiply':
        return num1 * num2
    
performOperation(2, 3)

def performOperation(num1, num2, operation='sum', message='Default message'):
    print(message)
    if operation == 'sum':
        return num1 + num2
    if operation == 'multiply':
        return num1 * num2
    
performOperation(2, 3, message='A new message!', operation='multiply')

# A new message!
```

*args

```py

    print(args)

    print(args)
```

**kwargs

```py
def performOperation(*args, **kwargs):
    print(args)
    print(kwargs)
performOperation(1,2,3, operation='sum')

# (1, 2, 3) - tuple кортеж
# {'operation': 'sum'} - словарь dictionary


import math
```

## 038-Variables and scope

 Видимость в простой функции
```py
def performOperation(*args, **kwargs):
    print(args)
    print(kwargs)

performOperation(1, 2, operation='sum')
# (1, 2) - кортеж
# {'operation': 'sum'} - словарь
```

Locals ()
```py
# Получить словарь всех переменных, которые были переданы

def performOperation(num1, num2, operation='sum'):
    print(locals())
    
performOperation(1, 2, operation='multiply')
# {'num1': 1, 'num2': 2, 'operation': 'multiply'}

```

globals()
```py
globals()
# будет выведен список глобальных переменных
```

Global and Local scope
```py
message = 'Some global data'

def function1(varA, varB):
    print(message)
    print(locals())
    
    
def function2(varC, varB):
    print(message)
    print(locals())
    
function1(1, 2)
function2(3, 4)

# Some global data
# {'varA': 1, 'varB': 2}
# Some global data
# {'varC': 3, 'varB': 4}

```

```py
message = 'Some global data'
varA = 2
def function1(varA, varB):
    message = 'Some local data'
    print(varA)
    print(message)
    print(locals())    
    
def function2(varC, varB):
    print(varA)
    print(message)
    print(locals())
    
function1(1, 2)
function2(3, 4)

# 1 - varA из локальной области, из функции
# Some local data - message из локальной области, из функции
# {'varA': 1, 'varB': 2, 'message': 'Some local data'}
# 2 - varA из глобальной области, вне функций
# Some global data - message из глобальной области, вне функций
# {'varC': 3, 'varB': 4}

```

Функция внутри другой функции
```py
def function1(varA, varB):
    message = 'Some local data'
    print(varA)
    def inner_function(varA, varB):
        print(f'inner_function local scope: {locals()}')
        
    inner_function(123, 456)
    

function1(1, 2) 

# 1
# inner_function local scope: {'varA': 123, 'varB': 456}

```

```py
def function1(varA, varB):
    message = 'Some local data'
    print(varA)
    def inner_function(varA, varB):
        print(f'inner_function local scope: {locals()}')
    
    print(locals())
    inner_function(123, 456)
    

function1(1, 2) 

# 1
# {'varA': 1, 'varB': 2, 'message': 'Some local data', 'inner_function': <function function1.<locals>.inner_function at 0x107b6af80>}
# inner_function local scope: {'varA': 123, 'varB': 456}
```
